Algorithmic Composition- Melody Mutator

Abstract:

Melody Mutator is an algorithm imitating the genome mutation process in parthenogenesis- a type of asexual reproduction. Given a melody sequence, for example, twinkle twinkle little star, the algorithm will repeat the same melody recursively but with minor increments or decrements in the musical parameters such as midi keynum, midi onset time, and duration. Eventually, you will hear a completely different sounding of the melody as if a white dwarf star becoming a supernova.

def MutateMidiNote(dur=0, amp=0, key=0, chan=0):
    dur = dur + choices([0, dur*0.1, -dur*0.1, dur*0.5, -dur*0.5], [70,1,1,1,1])[0]
    amp = amp + choices([0, amp*0.1, -amp*0.1, amp*0.5, -amp*0.5],[40,1,1,1,1])[0]
    key = key + choices([0, + 1, -1, + 2, -2],[30,1,1,1,1])[0]
    chan = choices([chan, 0, 1, 2, 3, 9],[40,1,1,1,1,1])[0]
    return (dur, amp, key, chan)
def mutate_pheno_1(q,notes,count):
    for j in range(count):
        for i in range(len(notes)):
            print(notes[i][2])
            note = MidiNote(time=q.now, dur=notes[i][0], amp=notes[i][1], key=notes[i][2], chan=notes[i][3])
            q.out.addevent(note)
            yield notes[i][0]
        notes = [MutateMidiNote(item[0],item[1],item[2],item[3]) for item in notes]


def get_melo(tone,keey):
    return [MutateMidiNote(0.3,0.5,item + keey,0) for item in tone]

def evolve():
    t0 = MidiSeq.metaseq(ins={0: AcousticGrandPiano, 1: Violin,2:Xylophone,3:Marimba})
    seq = MidiSeq()
    seq_2 = MidiSeq()
    seq_3 = MidiSeq()
    seq_4 = MidiSeq()
    queue = Scheduler(out=seq)
    queue_2 = Scheduler(out=seq_2)
    queue_3 = Scheduler(out=seq_3)
    queue_4 = Scheduler(out=seq_4)


    count = 20;
    tone = [60,60,67,67,69,69,67,67,65,65,64,64,62,62,60,60]
    A = mutate_pheno_1(queue, get_melo(tone,0),count)
    B = mutate_pheno_1(queue_2, get_melo(tone,12),count)
    C = mutate_pheno_1(queue_3, get_melo(tone,-12),count)
    D = mutate_pheno_1(queue_4, get_melo(tone,-24),count)
    #B = pheno_2(queue_2, chords, scales, key)
    queue.compose([0, A])
    queue_2.compose([0, B])
    queue_3.compose([0, C])
    queue_4.compose([0, D])
