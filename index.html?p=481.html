Algorithmic Composition- Octatonics

Abstract:

Octatonics is an algorithmic composition utilizing C octatonic scale to create aleatoric music for the game BGM. It is logically easy to implement under the musx framework in python but with interesting sounding.

def get_chords_octa(bar):
    octa_scale_1 = [0,1,3,4,6,7,9,10,0,1,3,4,6,7,9,10]
    octa_scale_2 = [0,2,3,5,6,8,9,11,0,2,3,5,6,8,9,11]
    chord_list = []
    for i in range(20):
        seed = choice(range(8))
        new = [octa_scale_1[seed], octa_scale_1[seed + choice([2,3])],\
             octa_scale_1[seed + choice([4,5])], octa_scale_1[seed + choice([6,7])]]
        new = [new[0]-12] + new
        chord_list.append(new)
    return (chord_list, octa_scale_1)

def middle_octa(q, chords, scales, keey):
    tempoo = 5
    cyc1 = cycle(rhythm("q q", tempo=tempoo))
    rhy1 = next(cyc1)
    rhy1 = next(cyc1)
    for chord in chords:
        rhy1 = next(cyc1)
        for k in chord:
            note = MidiNote(time=q.now, dur=rhy1, amp=choice([0.1,0.15]), key=keey - 12 +k, chan=0)
            q.out.addevent(note)
        yield rhy1
def melo_octa(q, chords, scales, keey):
    tempoo = 10
    cyc1 = cycle(rhythm("q q. e", tempo=tempoo))
    rhy1 = next(cyc1)
    rhy1 = next(cyc1)
    tone = [choice(scales) for i in range(40)]
    for t in tone:
        rhy1 = next(cyc1)
        note = MidiNote(time=q.now, dur=rhy1, amp=choice([0.2,0.3]), key=keey +12 +t, chan=0)
        q.out.addevent(note)
        yield rhy1

def octa():
    t0 = MidiSeq.metaseq(ins={0: AcousticGrandPiano, 1: Violin})
    seq = MidiSeq()
    seq_2 = MidiSeq()
    queue = Scheduler(out=seq)
    queue_2 = Scheduler(out=seq_2)
    # now getting the list of the chord
    bar = 32;temp = get_chords_octa(bar)
    chords = temp[0]; scales = temp[1]; key = 60 + 12 -1
    print(scales)
    queue.compose([0, middle_octa(queue, chords, scales, key)])
    queue_2.compose([0, melo_octa(queue_2, chords, scales, key)])
    #queue.compose(melody(queue, chords, scales, key))
    # print("yes")
    MidiFile("octa.mid", [t0,seq,seq_2]).write()
