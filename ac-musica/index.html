Algorithmic Composition- Musica

Abstract:

Muisca is an algorithmic composition that can generate functional four-part harmony based on the prescribed tessitura for each voice. Inside the algorithm, I implement multiple finite state machines with the probabilistic approach in different musical parameters controlling the transition of chord function(T, PD, D), scale degree(I, V, IV, etc.), and voice-leading rules(voice crossing, leap-recovery rule in melody, etc.). The aesthetic consideration of the generative algorithm is referred to a four-part chorale I composed, which is also attached below.

def s_elasticity_valid(l_notes, scope=4):
    l = l_notes 
    if scope == 4:
        # normalized
        # extract the last three
        l = [abs(pitch - l[0]) for pitch in l[0:4]]
        # Pythagorian rules for triangle
        if ((l[1] + l[2]) > l[3]) or ((l[1] + l[3]) > l[2]) or ((l[2] + l[3]) > l[1]):
            return True
        else:
            return False 
    elif scope == 8:
        l = [abs(pitch - l[0]) for pitch in l[0:8]]
         # Pythagorian rules for triangle for first 4 notes
        if ((l[1] + l[2]) > l[3]) or ((l[1] + l[3]) > l[2]) or ((l[2] + l[3]) > l[1]):
            if l[7]-l[0] < 5:
                return True
        else:
            return False

def motionize(voices):
    '''
    accepting the 4 voices generated by the generate CP, now bestow the motion and make
    the counterpoint alive!
    '''
    s_prime = voices[0]; a_prime = voices[1]; t_prime = voices[2]; b_prime = voices[3]
    # form_num = 4
    # # this is the piano roll for each voice
    # s = [0 for i in range(3*8*2*form_num)]

    # t = [0 for i in range(3*8*2*form_num)]
    # b = [0 for i in range(3*8*2*form_num)]
    
    chord_matrix = [] # this contains the whole piano roll of the corpus
    pc = [s_prime[0], a_prime[0], t_prime[0], b_prime[0]]
    chord_matrix.append(pc)
    for i in range(1, len(s_prime)):
        tc = [s_prime[i], a_prime[i], t_prime[i], b_prime[i]]
        inner_chord = return_inner_chord([pc,tc])
        chord_matrix.append(inner_chord[0])
        chord_matrix.append(inner_chord[1])
        chord_matrix.append(tc)
        pc = tc
    # now the chord_matrix would be 2D filled with the chord
    print(chord_matrix)
    return chord_matrix

def soprano(q,l,keey,tempoo):
    cyc1 = cycle(rhythm("q q", tempo=tempoo))
    rhy1 = next(cyc1)
    rhy1 = next(cyc1)
    #print(l)
    for t in l:
        rhy1 = next(cyc1)
        note = MidiNote(time=q.now, dur=rhy1, amp=choice([0.7]), key=keey +12 +t, chan=0)
        q.out.addevent(note)
        yield rhy1
