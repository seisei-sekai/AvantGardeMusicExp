Sonification – Epilepsy Harmonizer

Abstract:

Epilepsy Harmonization is a sonification algorithm depicting the contrast between brainwave during seizure and non-seizure. When a seizure occurs in the provided data, a series of atonal chords will be played against the tonally functional chords for non-seizure brainwaves. I set relative mean and variance thresholds on the magnitude to classify the seizure states of the given signals. (Though better to apply as a gaussian classifer)

    #=========================================================================
    print(len(label_list))

    # Now we will have 2 list, mean and std
    chunk_size = 20
    mean = []
    std = []
    for i in range(int(len(single_list)/chunk_size)):
        m = np.mean(abs_single_list[i*chunk_size:(i+1)*chunk_size])

def eeg_sonification(mean, std, srate):

    # mean is chord selection, std is the tempo rate
    # mean <200, 200< < 260, 260<600, 600<

    consonance_level_one = [[60-12, 60, 64, 67], \
                            [67-12, 59, 62, 67], \
                            [65-12, 69-12, 60, 67]]
    consonance_level_two = [[60-12, 60-1,62, 64, 67], \
                            [67-12, 59,60, 62,65, 67], \
                            [65-12, 69-12, 60,62, 67]]
    dissonance_level_one = [[60-12, 60-1,61,62, 64, 66, 67], \
                            [67-12, 59,60,61, 62,65,66, 67], \
                            [65-12, 69-12, 60,61,62,66, 67]]
    dissonance_level_two = [[60-12,58,59, 60-1,61,62, 64,65, 66, 67], \
                            [67-12, 58,59,60,61, 62,64,65,66, 67], \
                            [65-12, 69-12, 60,61,62,63,64,65,66, 67]]
    seed_list = [] # seed list is the chord for each second
    srate = int(srate / 8)
    for i in range(int(len(mean)/(srate))):
        m = np.mean(mean[i*srate:(i+1)*srate])
        if m <= 175:
            seed_list.append(0)
        elif m >175 and m <=230:
            seed_list.append(1)
        elif m > 230 and m <= 400:
            seed_list.append(2)
        elif m > 400:
            seed_list.append(3)
    play_list = []
    for i in range(len(seed_list)):
        if seed_list[i] == 0:
            play_list.append([random.choice(consonance_level_one)])
        elif seed_list[i] == 1:
            play_list.append([random.choice(consonance_level_two) for i in range(2)] )
        elif seed_list[i] == 2:
            play_list.append([random.choice(dissonance_level_one)  for i in range(4)] )
        elif seed_list[i] == 3:
            play_list.append([random.choice(dissonance_level_two)  for i in range(8)] )
    print(play_list)
    keynum_list = []
    rhythm_list = []
    
    # initializing keynum_list
    midi_event_count = 0
    for i in range(len(play_list)):
        for j in range(len(play_list[i])):
            keynum_list.append(play_list[i][j])
            for k in range(len(play_list[i][j])):
                midi_event_count +=1

    # initializing rhythm_list
    for i in range(len(play_list)):
        if len(play_list[i]) == 1:
            rhythm_list += [0.125]
        elif len(play_list[i]) == 2:
            rhythm_list += [0.125/2] * 2
        elif len(play_list[i]) == 3:
            rhythm_list += [0.125/4] * 4
        elif len(play_list[i]) == 4:
            rhythm_list += [0.125/8] * 8

    piano_pitch = keynum_list
    piano_rhythm = rhythm(rhythm_list, tempo = 60)
